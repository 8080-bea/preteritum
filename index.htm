        <div class="bg-white p-8 rounded-2xl max-w-md w-full text-center">

            <h2 class="text-3xl font-bold text-indigo-600 mb-4">Äventyret är slut!</h2>

            <p id="final-score" class="text-lg mb-6"></p>

    <script>

        const apiKey = ""; // Miljön hanterar detta

 

        const storySteps = [

            {

                text: "Erik ______ tidigt på morgonen och packade sin väska.",

                options: ["vaknar", "vaknade", "vaknat"],

                correct: 1,

                prompt: "A young explorer waking up in a cozy wooden cabin in the morning, sunlight coming through the window."

            },

            {

                text: "Han ______ en karta på bordet som visade vägen till berget.",

                options: ["hittade", "hittar", "hittat"],

                correct: 0,

                prompt: "An old weathered map lying on a wooden table next to a compass and a backpack."

            },

            {

                text: "Erik ______ ut ur huset och andades in den friska luften.",

                options: ["går", "gick", "gått"],

                correct: 1,

                prompt: "An explorer walking out of a small house into a bright green meadow at the edge of a forest."

            },

            {

                text: "Fåglarna ______ glatt uppe i de höga träden.",

                options: ["sjöng", "sjunger", "sjunga"],

                correct: 0,

                prompt: "Beautiful colorful birds sitting on tree branches in a lush forest."

            },

            {

                text: "Plötsligt ______ han ett mystiskt ljud bakom en buske.",

                options: ["hör", "hört", "hörde"],

                correct: 2,

                prompt: "A man looking curiously behind a large green bush in a dense forest."

            },

            {

                text: "En liten räv ______ fram och tittade på honom.",

                options: ["sprang", "springer", "sprungit"],

                correct: 0,

                prompt: "A cute small orange fox peeking out from behind a bush, curious expression."

            },

            {

                text: "Erik ______ en smörgås ur sin väska och gav en bit till räven.",

                options: ["tog", "tar", "tagit"],

                correct: 0,

                prompt: "Hand holding a piece of bread towards a small fox in a forest."

            },

            {

                text: "Sedan ______ de tillsammans mot det stora berget.",

                options: ["fortsätta", "fortsatte", "fortsätter"],

                correct: 1,

                prompt: "An explorer and a small fox walking together on a path towards a distant blue mountain."

            },

            {

                text: "Solen ______ varmt hela eftermiddagen.",

                options: ["skiner", "skinit", "sken"],

                correct: 2,

                prompt: "Bright sun over a beautiful mountain landscape with a clear sky."

            },

            {

                text: "När de kom fram ______ Erik äntligen den gömda skatten.",

                options: ["ser", "såg", "sett"],

                correct: 1,

                prompt: "An explorer finding a golden chest glowing inside a cave entrance at the foot of a mountain."

            }

        ];

 

        let currentStep = 0;

        let score = 0;

 

        const storyTextEl = document.getElementById('story-text');

        const optionsContainer = document.getElementById('options-container');

        const nextBtn = document.getElementById('next-btn');

        const feedbackEl = document.getElementById('feedback');

        const sceneImageEl = document.getElementById('scene-image');

        const loadingSpinner = document.getElementById('loading-spinner');

        const imagePlaceholder = document.getElementById('image-placeholder');

        const progressDots = document.getElementById('progress-dots');

 

        function initProgressDots() {

            progressDots.innerHTML = '';

            storySteps.forEach((_, i) => {

                const dot = document.createElement('div');

                dot.className = `h-2 w-4 rounded-full transition-colors ${i === 0 ? 'bg-white' : 'bg-indigo-400'}`;

                dot.id = `dot-${i}`;

                progressDots.appendChild(dot);

            });

        }

 

        async function generateImage(prompt) {

            loadingSpinner.classList.remove('hidden');

            imagePlaceholder.classList.add('hidden');

            sceneImageEl.classList.add('hidden');

 

            try {

                const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/imagen-4.0-generate-001:predict?key=${apiKey}`, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify({

                        instances: [{ prompt: prompt + " digital art style, vibrant colors, child friendly" }],

                        parameters: { sampleCount: 1 }

                    })

                });

 

                if (!response.ok) throw new Error('Image generation failed');

                const result = await response.json();

                const b64 = result.predictions[0].bytesBase64Encoded;

                sceneImageEl.src = `data:image/png;base64,${b64}`;

                sceneImageEl.classList.remove('hidden');

            } catch (error) {

                console.error(error);

                imagePlaceholder.classList.remove('hidden');

                imagePlaceholder.innerHTML = '<span class="text-xs text-red-400 p-4">Kunde inte ladda bild, men spelet fortsätter!</span>';

            } finally {

                loadingSpinner.classList.add('hidden');

            }

        }

 

        function loadStep() {

            const step = storySteps[currentStep];

           

            // UI Reset

            feedbackEl.classList.add('hidden');

            nextBtn.classList.add('hidden');

           

            // Text Update

            storyTextEl.innerHTML = `<p class="fade-in">${step.text.replace('______', '<span class="border-b-2 border-indigo-400 px-4 py-1 text-indigo-600 font-bold">?</span>')}</p>`;

           

            // Options Update

            optionsContainer.innerHTML = '';

            step.options.forEach((opt, index) => {

                const btn = document.createElement('button');

                btn.className = "text-left p-4 rounded-xl border-2 border-slate-200 hover:border-indigo-400 hover:bg-slate-50 transition-all font-medium text-lg";

                btn.textContent = opt;

                btn.onclick = () => checkAnswer(index, btn);

                optionsContainer.appendChild(btn);

            });

 

            // Update progress

            document.querySelectorAll('#progress-dots div').forEach((dot, i) => {

                dot.classList.toggle('bg-white', i === currentStep);

                dot.classList.toggle('bg-indigo-400', i !== currentStep);

            });

 

            // Generate Image

            generateImage(step.prompt);

        }

 

        function checkAnswer(selectedIndex, btn) {

            const step = storySteps[currentStep];

            const buttons = optionsContainer.querySelectorAll('button');

           

            // Disable all buttons

            buttons.forEach(b => b.disabled = true);

 

            if (selectedIndex === step.correct) {

                btn.classList.add('correct');

                feedbackEl.textContent = "Snyggt! Rätt form av verbet.";

                feedbackEl.className = "mt-6 p-4 rounded-lg text-emerald-700 bg-emerald-50 text-center font-medium block";

                storyTextEl.innerHTML = `<p class="fade-in">${step.text.replace('______', `<span class="text-emerald-600 font-bold underline">${step.options[selectedIndex]}</span>`)}</p>`;

                score++;

            } else {

                btn.classList.add('wrong');

                buttons[step.correct].classList.add('correct');

                feedbackEl.textContent = `Nära! Det rätta ordet är "${step.options[step.correct]}".`;

                feedbackEl.className = "mt-6 p-4 rounded-lg text-red-700 bg-red-50 text-center font-medium block";

                storyTextEl.innerHTML = `<p class="fade-in">${step.text.replace('______', `<span class="text-red-500 font-bold line-through">${step.options[selectedIndex]}</span> <span class="text-emerald-600 font-bold underline">${step.options[step.correct]}</span>`)}</p>`;

            }

 

            nextBtn.classList.remove('hidden');

        }

 

        nextBtn.onclick = () => {

            currentStep++;

            if (currentStep < storySteps.length) {

                loadStep();

            } else {

                document.getElementById('final-score').textContent = `Du fick ${score} av ${storySteps.length} rätt! Bra jobbat.`;

                document.getElementById('results-modal').classList.remove('hidden');

            }

        };

 

        // Start

        initProgressDots();

        loadStep();

    </script>
